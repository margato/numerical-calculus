Função retorna coeficiente linear e angular da reta de regressão, respectivamente¶

In [10]:
def regression(data):  # data: (x, y)
    alpha = beta = delta = gamma = 0.0
    for year, population in data:
        ln_population = math.log(population)
        gamma += year
        delta += year ** 2
        beta += year * ln_population
        alpha += ln_population
    n = len(data)
    det1 = n * delta - gamma ** 2
    det2 = alpha * delta - gamma * beta
    det3 = n * beta - alpha * gamma
    a = det2 / det1
    b = det3 / det1
    return a, b

def get_expression(a, b, points):
    return a + b * points, f"{'%.3f'%(a)} + {'%.3f'%(b)} * x"
