%matplotlib inline
import numpy as np
import pandas as pd
import math
import matplotlib.pyplot as plt

def f(x):
    # f(x) = e^(-x)
    return math.exp(-x**2)

a = float(input("A: "))
b = float(input("B: "))
n = int(input("Iteraes: "))
n = n + 1
h = float((b - a) / n)

ys = []
xs = np.linspace(a, b, n, endpoint=True).tolist()
print(f"Intervalo: [{a}, {b}], iteraes: {n - 1}, passo: {h}")
for xi in xs:
    ys.append(f(xi))
points = {
    'x': xs,
    'y': ys
}
df = pd.DataFrame(points)
df.index = df.index + 1

Funo que retorna a rea da curva pelo mtodo dos trapzios

In [30]:
def trapezes(h, ys):
    y_sum = sum(ys[1:(n - 1)])
    return (h/2) * (ys[0] + ys[-1] + 2 * y_sum)
